    <script>
        const searchInput = document.getElementById('searchInput');
        const clearSearchBtn = document.getElementById('clearSearch');
        const ceapsGrid = document.getElementById('ceapsGrid');
        const bhGrid = document.getElementById('bhGrid');
        const noProviderGrid = document.getElementById('noProviderGrid');

        const ceapsCount = document.getElementById('ceapsCount');
        const bhCount = document.getElementById('bhCount');
        const noProviderCount = document.getElementById('noProviderCount');

        const allExams = [];

        function collectExams(grid, type) {
            Array.from(grid.children).forEach(item => {
                allExams.push({
                    element: item,
                    text: item.textContent.toLowerCase(),
                    type: type
                });
            });
        }

        collectExams(ceapsGrid, 'ceaps');
        collectExams(bhGrid, 'bh');
        collectExams(noProviderGrid, 'noProvider');

        function updateCounts() {
            const visibleCeaps = allExams.filter(exam => exam.type === 'ceaps' && exam.element.style.display !== 'none').length;
            const visibleBh = allExams.filter(exam => exam.type === 'bh' && exam.element.style.display !== 'none').length;
            const visibleNoProvider = allExams.filter(exam => exam.type === 'noProvider' && exam.element.style.display !== 'none').length;

            ceapsCount.textContent = visibleCeaps;
            bhCount.textContent = visibleBh;
            noProviderCount.textContent = visibleNoProvider;
        }

        searchInput.addEventListener('keyup', function() {
            const searchTerm = searchInput.value.toLowerCase();
            let hasVisibleResults = false;

            allExams.forEach(exam => {
                if (exam.text.includes(searchTerm)) {
                    exam.element.style.display = 'block';
                    hasVisibleResults = true;
                } else {
                    exam.element.style.display = 'none';
                }
            });

            updateCounts();
            clearSearchBtn.style.display = searchTerm ? 'block' : 'none';
        });

        function clearSearch() {
            searchInput.value = '';
            allExams.forEach(exam => {
                exam.element.style.display = 'block';
            });
            updateCounts();
            clearSearchBtn.style.display = 'none';
        }

        clearSearchBtn.addEventListener('click', clearSearch);

        // Initial count update
        updateCounts();
    </script>
