        const searchInput = document.getElementById('searchInput');

        const allExams = [];

        function collectExams(grid, type) {
            Array.from(grid.children).forEach(item => {
                allExams.push({
                    element: item,
                    text: item.textContent.toLowerCase(),
                    type: type
                });
            });
        }

        collectExams(mediumComplexityGrid, 'medium');
        collectExams(highCostGrid, 'highCost');
        collectExams(specialGrid, 'special');
        collectExams(verificationGrid, 'verification');

        function updateCounts() {
            const visibleMedium = allExams.filter(exam => exam.type === 'medium' && exam.element.style.display !== 'none').length;
            const visibleHighCost = allExams.filter(exam => exam.type === 'highCost' && exam.element.style.display !== 'none').length;
            const visibleSpecial = allExams.filter(exam => exam.type === 'special' && exam.element.style.display !== 'none').length;
            const visibleVerification = allExams.filter(exam => exam.type === 'verification' && exam.element.style.display !== 'none').length;

            mediumCount.textContent = visibleMedium;
            highCostCount.textContent = visibleHighCost;
            // Assuming 'Sem Prestador' count is derived from specific items or a separate list not explicitly in the HTML structure shown.
            // For now, setting it to 0 or a placeholder. If there are specific 'Sem Prestador' items, they need to be filtered.
            noProviderCount.textContent = allExams.filter(exam => exam.text.includes('sem prestador') && exam.element.style.display !== 'none').length;
            totalCount.textContent = visibleMedium + visibleHighCost + visibleSpecial + visibleVerification;

        searchInput.addEventListener('keyup', function() {
            const searchTerm = searchInput.value.toLowerCase();
            let hasVisibleResults = false;

            allExams.forEach(exam => {
                if (exam.text.includes(searchTerm)) {
                    exam.element.style.display = 'block';
                    hasVisibleResults = true;
                } else {
                    exam.element.style.display = 'none';
                }
            });

            updateCounts();
            clearSearchBtn.style.display = searchTerm ? 'block' : 'none';
        });

        function clearSearch() {
            searchInput.value = '';
            allExams.forEach(exam => {
                exam.element.style.display = 'block';
            });
            updateCounts();
            clearSearchBtn.style.display = 'none';
        }

        // Initial count update
        updateCounts();
    </script>
</body>
