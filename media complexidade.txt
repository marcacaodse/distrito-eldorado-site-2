    <script>
        document.getElementById('currentYear').textContent = new Date().getFullYear();

        const searchInput = document.getElementById('searchInput');
        const examItems = document.querySelectorAll('.exam-item');
        const noResultsMessage = document.getElementById('noResultsMessage');
        const mediumCount = document.getElementById('mediumCount');
        const highCostCount = document.getElementById('highCostCount');
        const noProviderCount = document.getElementById('noProviderCount');
        const totalCount = document.getElementById('totalCount');

        const initialCounts = {
            medium: 41,
            highCost: 16,
            noProvider: 21,
            total: 81
        };

        function updateCounts(visibleMedium, visibleHighCost, visibleNoProvider) {
            mediumCount.textContent = visibleMedium;
            highCostCount.textContent = visibleHighCost;
            noProviderCount.textContent = visibleNoProvider;
            totalCount.textContent = visibleMedium + visibleHighCost + visibleNoProvider;
        }

        searchInput.addEventListener('keyup', function() {
            const searchTerm = searchInput.value.toLowerCase();
            let anyVisible = false;
            let currentMedium = 0;
            let currentHighCost = 0;
            let currentNoProvider = 0;

            examItems.forEach(item => {
                const text = item.textContent.toLowerCase();
                if (text.includes(searchTerm)) {
                    item.style.display = 'block';
                    anyVisible = true;
                    if (item.closest('#mediumComplexityGrid')) {
                        currentMedium++;
                    } else if (item.closest('#highCostGrid')) {
                        currentHighCost++;
                    } else if (item.closest('#noProviderGrid')) {
                        currentNoProvider++;
                    }
                } else {
                    item.style.display = 'none';
                }
            });

            updateCounts(currentMedium, currentHighCost, currentNoProvider);

            if (anyVisible) {
                noResultsMessage.classList.add('hidden');
            } else {
                noResultsMessage.classList.remove('hidden');
            }
        });

        function clearSearch() {
            searchInput.value = '';
            examItems.forEach(item => {
                item.style.display = 'block';
            });
            updateCounts(initialCounts.medium, initialCounts.highCost, initialCounts.noProvider);
            noResultsMessage.classList.add('hidden');
        }
    </script>
